 StreamReader f = new StreamReader(Server.MapPath("katalog.txt"));           
 List<CD> sviCDovi = CD.procitaj(f);

 foreach (CD k in sviCDovi)
{
if ((txtIzvodjac.Text == "" || k.izvodjac.Contains(txtIzvodjac.Text)) &&
 (txtAlbum.Text == "" || k.album.Contains(txtAlbum.Text)) && 
(ddlZanr.SelectedValue == "сви жанрови" || (k.zanr.Contains(ddlZanr.SelectedValue)))&& 
(ddlGodina.SelectedValue == "све године" || (k.godina.ToString() == ddlGodina.SelectedValue))&& 
(txtIzdavac.Text == "" || k.izdavac.Contains(txtIzdavac.Text)))  
              {                   
 izabraniCDovi.Add(k);        
        }            }         
   f.Close();
            gvIzabrani.DataSource = izabraniCDovi;
gvIzabrani.DataBind();} } }

CD.cs 

using System; 
using System.Collections.Generic; 
using System.IO; 
namespace А13CDKatalog{   
 public class CD{
public string izvodjac { get; set; }        
public string album { get; set; }        
public string zanr { get; set; }        
public int godina { get; set; }        
public string izdavac { get; set; }        
public string slika { get; set; }        
public CD(string izvodjac, string album, string zanr, int godina, string izdavac, string slika)
{            
this.izvodjac = izvodjac;            
this.album = album;            
this.zanr = zanr;            
this.godina = godina;           
this.izdavac = izdavac;            
this.slika = slika;        
}        
public static List<CD> procitaj(StreamReader f){           
 List<CD> sviKorisnici = new List<CD>();            
while (!f.EndOfStream){               
 string[] sviPodaci = f.ReadLine().Split('|');               
 CD k = new CD(sviPodaci[0], sviPodaci[1], sviPodaci[2],                        
Convert.ToInt32(sviPodaci[3]), sviPodaci[4], sviPodaci[5]);                
sviKorisnici.Add(k);     
       }           
 return sviKorisnici;     
   }    } }
