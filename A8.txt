Proizvod.cs 

using System; 
using System.Collections.Generic; 
using System.Linq; using System.Web; 
using System.IO; namespace A8{    
public class Proizvod{        
public string sifra { get; set; }       
 public string naziv { get; set; }        
public string proizvodjac { get; set; }        
public string ram { get; set; }        
public string procesor { get; set; }        
public string kamera { get; set; }        
public double ekran { get; set; }        
public string slika { get; set; }        
public int cena { get; set; }        
public Proizvod(string sifra, string naziv, string proizvodjac, string ram, string procesor, string kamera, double ekran, string slika, int cena){            this.sifra = sifra;            
this.naziv = naziv;            
this.proizvodjac = proizvodjac;            
this.ram = ram;            
this.procesor = procesor;            
this.kamera = kamera;            
this.ekran = ekran;            
this.slika = slika;            
this.cena = cena;        
}        
public static List<Proizvod> procitaj(StreamReader f){           
 List<Proizvod> proizvodi = new List<Proizvod>();            
while (!f.EndOfStream){                
string linija = f.ReadLine();                
Proizvod p = new Proizvod(linija.Substring(0, 6), linija.Substring(6, 25), linija.Substring(31, 20), linija.Substring(51, 5).Trim(), linija.Substring(56, 15), linija.Substring(71, 10), Convert.ToDouble(linija.Substring(81, 5)), linija.Substring(91, 30), Convert.ToInt32(linija.Substring(121, 10)));                proizvodi.Add(p);          
  }            
return proizvodi; } } }
