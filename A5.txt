funkcionalnost.js 

function kreirajSedista()
{
 var br=2;
while(br<=52)
{var red=document.createElement('tr'); 
document.getElementById("tabela").appendChild(red);
for(var i=0;i<4;i++)
{ 
if(br>52) return; 
//крај рада функције за "седиште 53" 
var dugme=document.createElement('button');
//ново седиште 
dugme.innerHTML=br;
//упис броја седишта на дугмић 
dugme.setAttribute("class","sediste");
//ознака класе 
dugme.setAttribute("id",br);
//број седишта је id дугмића 
dugme.setAttribute("onclick","klik("+br+")");
red.appendChild(dugme);
//додавање дугмића у ред седишта 
br++;} } } 
function klik(br){
//када се кликне на неко "седиште" у табели...    
document.getElementById("broj").value=br; } 
function obojSediste()
{var brSedista=document.getElementById("broj").value;
 document.getElementById(brSedista).style.backgroundColor="red";
document.getElementById(brSedista).disabled=true;}
